import js from pyodide.ffi import create_proxy import random import time import json from datetime import datetime # ========================================== # Assets: SVG definitions for 6 distinct items # ========================================== SVG_ITEMS = { "蘋果": '

', "西瓜": '

' } # ========================================== # Audio synthesis for positive/negative tones # ========================================== def play_tone(freq, type, duration): try: audio_ctx = js.window.AudioContext.new() if hasattr(js.window, 'AudioContext') else js.window.webkitAudioContext.new() oscillator = audio_ctx.createOscillator() gain_node = audio_ctx.createGain() oscillator.type = type oscillator.frequency.value = freq oscillator.connect(gain_node) gain_node.connect(audio_ctx.destination) oscillator.start() gain_node.gain.exponentialRampToValueAtTime(0.00001, audio_ctx.currentTime + duration) oscillator.stop(audio_ctx.currentTime + duration) except Exception as e: print("Audio not supported or error:", e) def play_success(): play_tone(600, 'sine', 0.1) js.setTimeout(create_proxy(lambda: play_tone(800, 'sine', 0.2)), 100) def play_error(): play_tone(300, 'triangle', 0.3) # ========================================== # Game State # ========================================== class GameState: def __init__(self): self.round_start_time = 0 self.target_name = "" self.target_index = -1 self.total_correct = 0 self.num_items = 5 # 5 items spread across screen self.timer_id = None self.hint_timer_id = None self.stats = self.load_stats() self.today_date = datetime.now().strftime("%Y-%m-%d") self.locked = False # Prevent clicks during transitions if self.today_date not in self.stats: self.stats[self.today_date] = {"count": 0, "total_time": 0.0} def load_stats(self): try: data = js.window.localStorage.getItem("rehab_scan_data") if data: return json.loads(data) except Exception: pass return {} def save_stats(self): try: js.window.localStorage.setItem("rehab_scan_data", json.dumps(self.stats)) self.update_stats_ui() except Exception: pass def update_stats_ui(self): today_data = self.stats.get(self.today_date, {"count": 0, "total_time": 0.0}) js.document.getElementById("today-count").innerText = str(today_data["count"]) # Build last 7 days history container = js.document.getElementById("stats-table-container") container.innerHTML = "" dates = sorted(self.stats.keys())[-7:] for t_date in dates: data = self.stats[t_date] count = data["count"] avg = 0 if count == 0 else data["total_time"] / count div = js.document.createElement("div") div.className = "stat-item" display_date = t_date[-5:] # MM-DD div.innerText = f"{display_date}: {count}題 (平均 {avg:.1f}s)" container.appendChild(div) state = GameState() # ========================================== # Core Game Logic # ========================================== def render_round(): state.locked = False clear_timers() js.document.getElementById("left-boundary").classList.remove("boundary-flashing") js.document.getElementById("hint-message").style.opacity = "0" container = js.document.getElementById("items-container") container.innerHTML = "" # Target Selection names = list(SVG_ITEMS.keys()) state.target_name = random.choice(names) names.remove(state.target_name) distractors = random.sample(names, state.num_items - 1) # 80% left side logic if random.random() < 0.80: state.target_index=0 else: state.target_index=random.randint(1, state.num_items - 1) items=distractors[:] items.insert(state.target_index, state.target_name) # Build DOM for i, name in enumerate(items): div=js.document.createElement("div") div.className="item" div.id=f"item-{i}" div.innerHTML=SVG_ITEMS[name] # Using a closure factory to bind the correct index def make_handler(idx): def on_click(evt): if not state.locked: handle_click(idx) return create_proxy(on_click) div.addEventListener("click", make_handler(i)) container.appendChild(div) # Update Instruction js.document.getElementById("instruction").innerText=f"請找出 最左邊 的 【{state.target_name}】" state.round_start_time=time.time() # Set 5s inactivity timer state.timer_id=js.setTimeout(create_proxy(check_inactivity), 5000) def handle_click(clicked_index): if clicked_index==state.target_index: # Correct guess logic state.locked=True clear_timers() play_success() elapsed=time.time() - state.round_start_time # Stop flashing js.document.getElementById("left-boundary").classList.remove("boundary-flashing") item_el=js.document.getElementById(f"item-{clicked_index}") if item_el: item_el.classList.remove("item-flashing") item_el.style.transform="scale(1.3)" item_el.style.zIndex="100" # Update metrics today=state.stats[state.today_date] today["count"] +=1 today["total_time"] +=elapsed state.total_correct +=1 state.save_stats() if state.total_correct % 10==0: js.setTimeout(create_proxy(show_reward), 600) else: js.setTimeout(create_proxy(render_round), 800) else: # Incorrect guess logic play_error() show_hint("再往左邊找找看") def show_hint(msg): hint=js.document.getElementById("hint-message") hint.innerText=msg hint.style.opacity="1" if state.hint_timer_id is not None: js.clearTimeout(state.hint_timer_id) def hide_hint(): hint.style.opacity="0" state.hint_timer_id=js.setTimeout(create_proxy(hide_hint), 2000) def check_inactivity(): if state.target_index==0: # Prompt visual neglect flashes js.document.getElementById("left-boundary").classList.add("boundary-flashing") item=js.document.getElementById("item-0") if item: item.classList.add("item-flashing") def show_reward(): modal=js.document.getElementById("reward-modal") modal.style.display="flex" flowers=["🌹", "🌺" , "🌸" , "🌻" , "🌷" ] js.document.getElementById("flower").innerText=random.choice(flowers) def hide_reward(): modal.style.display="none" render_round() js.setTimeout(create_proxy(hide_reward), 4000) def clear_timers(): if state.timer_id is not None: js.clearTimeout(state.timer_id) state.timer_id=None # Boot app state.update_stats_ui() render_round()
